

 correct TestDB.java:

import java.sql.Connection;
import java.sql.DriverManager;

public class TestDB {
    public static void main(String[] args) {
        try {
            Class.forName("com.mysql.cj.jdbc.Driver");
            Connection conn = DriverManager.getConnection(
                "jdbc:mysql://mysql-service1:3306/db1", "root", "Login%12345");
            System.out.println("Connected!");
            conn.close();
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}


Then compile and run:

# Compile
javac -cp "/usr/local/tomcat/lib/mysql-connector-j-9.1.0.jar:." TestDB.java

# Run
java -cp ".:/usr/local/tomcat/lib/mysql-connector-j-9.1.0.jar" TestDB


No need to include servlet-api.jar here because this class is not a servlet, just a standalone Java test program.
